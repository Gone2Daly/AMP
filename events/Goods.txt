country_event = {

	id = 1070

	trigger = {
		invention=cotton_gin
		any_owned_province = {
		   
		   trade_goods=cotton
		}
		NOT={has_country_flag=cotton_gin_flag}
	}

	mean_time_to_happen =  {
		months = 24
	}

	title = "Изобретение коттон-джина и внедрение его в $COUNTRY_ADJ$ хлопкоробстве!"
	desc = "Наша хлопкозаготовительная промышленность переживает невероятный бум! Недавнее изобретение хлопкоочистительной машины и её внедрение произвело революцию в обработке хлопка!"
	picture = "cotton_gin"

	option = {
		name = "Хлопок озолотит нас!"
                set_country_flag=cotton_gin_flag
		any_owned = {
		   limit = {
		      
		      trade_goods=cotton
                      NOT={has_province_modifier=cotton_gin_modifier}
		   }
		   add_province_modifier={name=cotton_gin_modifier duration=-1}
		}
	}
}
